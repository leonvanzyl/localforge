<project_specification>
  <project_name>LocalForge</project_name>

  <overview>
    LocalForge is a long-running autonomous coding harness that uses free, local AI models (via LM Studio) to build applications on autopilot. Users describe what they want to build, and LocalForge breaks it into features, tracks them on a kanban board, and deploys coding agents to implement and test them one by one — all powered by local models running on the user's own hardware. The target audience is vibe coders who want to describe an app and watch it get built without needing cloud-based AI services.
  </overview>

      <framework>Next.js 16 with React 19</framework>

      <drag_and_drop>dnd-kit or similar for kanban drag-and-drop</drag_and_drop>

    <backend>

      <supported>LM Studio only (MVP)</supported>

  <database_schema>
    <tables>
      <projects>
        - id (integer, primary key, auto-increment)
        - name (text, not null)
        - description (text, nullable)
        - folder_path (text, not null, unique)
        - status (text: active, completed, archived)
        - created_at (timestamp, default now)
        - updated_at (timestamp, default now)
      </projects>

      <features>
        - id (integer, primary key, auto-increment)
        - project_id (integer, foreign key -> projects.id)
        - title (text, not null)
        - description (text, nullable)
        - acceptance_criteria (text, nullable)
        - status (text: backlog, in_progress, completed)
        - priority (integer, default 0)
        - category (text: functional, style)
        - created_at (timestamp, default now)
        - updated_at (timestamp, default now)
      </features>

      <feature_dependencies>
        - id (integer, primary key, auto-increment)
        - feature_id (integer, foreign key -> features.id)
        - depends_on_feature_id (integer, foreign key -> features.id)
      </feature_dependencies>

      <agent_sessions>
        - id (integer, primary key, auto-increment)
        - project_id (integer, foreign key -> projects.id)
        - feature_id (integer, foreign key -> features.id, nullable)
        - session_type (text: coding, bootstrapper)
        - status (text: in_progress, completed, failed, terminated)
        - started_at (timestamp, default now)
        - ended_at (timestamp, nullable)
      </agent_sessions>

      <agent_logs>
        - id (integer, primary key, auto-increment)
        - session_id (integer, foreign key -> agent_sessions.id)
        - feature_id (integer, foreign key -> features.id, nullable)
        - message (text, not null)
        - message_type (text: info, action, error, screenshot, test_result)
        - screenshot_path (text, nullable)
        - created_at (timestamp, default now)
      </agent_logs>

      <chat_messages>
        - id (integer, primary key, auto-increment)
        - session_id (integer, foreign key -> agent_sessions.id)
        - role (text: user, assistant)
        - content (text, not null)
        - created_at (timestamp, default now)
      </chat_messages>

      <settings>
        - id (integer, primary key, auto-increment)
        - project_id (integer, foreign key -> projects.id, nullable for global settings)
        - key (text, not null)
        - value (text, not null)
      </settings>
    </tables>
  </database_schema>

  <api_endpoints_summary>
    <projects>
      - GET /api/projects (list all projects)
      - POST /api/projects (create new project)
      - GET /api/projects/:id (get project details)
      - PATCH /api/projects/:id (update project)
      - DELETE /api/projects/:id (delete project, optional file removal)
    </projects>
    <features>
      - GET /api/projects/:id/features (list features for project)
      - POST /api/projects/:id/features (create feature)
      - PATCH /api/features/:id (update feature - title, description, priority, status)
      - DELETE /api/features/:id (delete feature)
      - PATCH /api/features/:id/status (move feature between columns)
      - PUT /api/projects/:id/features/reorder (reorder features / update priorities)
    </features>
    <dependencies>
      - GET /api/features/:id/dependencies (get dependencies for feature)
      - POST /api/features/:id/dependencies (add dependency)
      - DELETE /api/features/:featureId/dependencies/:depId (remove dependency)
    </dependencies>
    <orchestrator>
      - POST /api/projects/:id/orchestrator/start (start orchestrator for project)
      - POST /api/projects/:id/orchestrator/stop (force-stop orchestrator)
      - GET /api/projects/:id/orchestrator/status (get orchestrator status)
    </orchestrator>
    <agent>
      - GET /api/agent/stream/:sessionId (SSE stream for live agent output)
      - GET /api/agent/logs/:featureId (get agent logs for a feature)
      - GET /api/agent/screenshots/:logId (serve screenshot image)
    </agent>
    <bootstrapper>
      - POST /api/projects/:id/bootstrapper/start (start AI conversation session)
      - POST /api/bootstrapper/:sessionId/message (send message in conversation)
      - GET /api/bootstrapper/:sessionId/messages (get conversation history)
      - POST /api/bootstrapper/:sessionId/complete (mark conversation complete)
    </bootstrapper>
    <settings>
      - GET /api/settings (get global settings)
      - PUT /api/settings (update global settings)
      - GET /api/projects/:id/settings (get project settings)
      - PUT /api/projects/:id/settings (update project settings)
    </settings>
  </api_endpoints_summary>
